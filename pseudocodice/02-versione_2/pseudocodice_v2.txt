void processFiles() {
	for (File file : fileList) {
		processFile(file);
	}
}

void processFile(File file) {
	// Ci prendiamo lista metodi in proc ALL
	LinkedList<String> methodsList = CcsManager.getMethodsListInProcAll(filePath);

	for (String method : methodsList) {

	}
}
